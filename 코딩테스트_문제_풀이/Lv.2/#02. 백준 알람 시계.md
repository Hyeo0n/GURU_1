```
def solution():
    # 사용자한테 시(H)와 분(M)을 입력받음
    H, M = map(int, input("Enter hours and minutes: ").split())
    
    # 분이 45분보다 작을 때 -> 45분 전의 시간이 현재 시간보다 전날로 넘어가야 함
    if M < 45:
        # 시가 0일 경우 (자정일 때) -> 하루가 바뀌니까 23시가 됨
        if H == 0:
            H = 23  # 0시에서 1시간 전은 23시 (전날의 마지막 시간)
            M += 60  # 분을 60분 추가하여 전체 분으로 계산 -> 0시부터 지금까지 총 분 계산하려면
        else:
            # 시가 0이 아닐 경우 (0보다 큰 시) -> 걍 일반적으로 계산하면 됨
            H -= 1  # 한 시간 전으로 변경 -> 한 시간 전으로 변경
            M += 60  # 분을 60분 추가하여 전체 분으로 계산 -> 60분을 추가해 전체 분으로 변환
    
    # 45분을 뺀 시간을 출력
    print(H, M - 45)

# 함수 호출
solution()
```
