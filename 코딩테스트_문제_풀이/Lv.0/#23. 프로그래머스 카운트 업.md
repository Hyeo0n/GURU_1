def solution(start_num, end_num):
    # start_num부터 end_num까지의 숫자를 리스트로 생성함
    return list(range(start_num, end_num + 1))

# 예제 사용
def main():
    print(solution(3, 10))  # 출력: [3, 4, 5, 6, 7, 8, 9, 10]

# 함수 호출
main()
