```
def solution(num1, num2):
    # num1을 num2로 나눈 나머지를 반환함
    return num1 % num2

# 예시 테스트
def main():
    print(solution(10, 3))  # 출력: 1
    print(solution(20, 4))  # 출력: 0
    print(solution(7, 5))   # 출력: 2

# 함수 호출
main()
```
