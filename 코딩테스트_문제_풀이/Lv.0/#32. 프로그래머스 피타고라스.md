def solution():
    # a와 c를 각각 입력받음
    a = int(input("Enter the value of a: "))
    c = int(input("Enter the value of c: "))

    # 피타고라스 정리에 따라 b^2를 계산함
    b_square = c**2 - a**2

    # 결과 출력
    print(b_square)

# 함수 호출
solution()
