```
def solution(num1, num2):
    # num1과 num2가 같은지 비교함
    if num1 == num2:
        return 1
    else:
        return -1

# 예제 사용
def main():
    print(solution(5, 5))  # 출력: 1
    print(solution(3, 4))  # 출력: -1

# 함수 호출
main()
```
