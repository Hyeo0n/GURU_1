def solution(rny_string):
    # 'm'을 'rn'으로 대체함
    return rny_string.replace('m', 'rn')

# 예제 사용
def main():
    print(solution("map"))  # 출력: "rnap"
    print(solution("maximum"))  # 출력: "rnaxirnurn"

# 함수 호출
main()
