# 사용자한테 문자열 입력받음
word = input("Enter a string: ")

# 문자열의 각 인덱스에 대해 반복
for i in range(len(word) // 2):
    # 문자열의 i번째 문자와 대칭 위치의 문자를 비교
    if word[i] != word[-1 - i]:
        # 문자가 일치하지 않으면 0을 출력하고 프로그램을 종료
        print(0)
        exit(0)

# 모든 문자가 대칭 위치와 일치하면 1을 출력
print(1)
